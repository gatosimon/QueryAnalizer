SELECT *
FROM (
    SELECT 
        M.COD_COMUNA AS COMUNA,
        M.COD_JURISDICCION AS INSTANCIA,
        M.COD_INSTANCIA AS JUZGADONRO,
        M."AÑO_DEMANDA",
        M.NRO_JUICIO,
        M.FECHA_DEMANDA,
        E.COD_ESTADO,
        MA.DENOAM AS NOM_ESTADO,
        E.DESCRIPCION AS COMENTARIO,
        M.CUIJ_J,
        E.FECHATRAMITEJUIC AS FECHA_TRAMITE,
        ROW_NUMBER() OVER(
            PARTITION BY M.COD_COMUNA, M.COD_JURISDICCION, M.COD_INSTANCIA, M."AÑO_DEMANDA", M.NRO_JUICIO
            ORDER BY E.FECHATRAMITEJUIC DESC
        ) AS RN
    FROM M10NAPLEG M
    JOIN M10ESTJU E
      ON E.CODCOMUNA = M.COD_COMUNA
     AND E.JURISDICCION = M.COD_JURISDICCION
     AND E.INSTANCIA = M.COD_INSTANCIA
     AND E.ANODEMANDA = M."AÑO_DEMANDA"
     AND E.NROJUICIO = M.NRO_JUICIO
    JOIN M00TABMA MA
      ON MA.CODPRIN = 452 
     AND MA.SUBCOD = E.COD_ESTADO
    WHERE M.NRO_JUICIO BETWEEN 1 AND 99999999
      AND M.FECHA_DEMANDA BETWEEN DATE('2020-01-01') AND DATE('2025-10-27')
      AND (M.COD_CANCELAC = '' OR M.COD_CANCELAC IS NULL)
      AND M.FECHA_CANCEL = DATE('1900-01-01')
      AND M.TIPOJUICIO = ''
      AND (CURRENT DATE - E.FECHATRAMITEJUIC) >= 10
      AND E.FECHA_BAJA = DATE('1900-01-01')
) T
WHERE RN = 1;

E.JURISDICCION = M.COD_JURISDICCION AND 
E.INSTANCIA = M.COD_INSTANCIA AND 
E.ANODEMANDA = M.AÑO_DEMANDA AND 
E.NROJUICIO = M.NRO_JUICIO 
JOIN M00TABMA MA 
ON MA.CODPRIN = 452 AND 
MA.SUBCOD = E.COD_ESTADO 
WHERE 
M.NRO_JUICIO BETWEEN 1 AND 99999999 AND 
M.FECHA_DEMANDA BETWEEN '15/09/2018' AND '29/10/2025' AND 
M.COD_CANCELAC = '' AND 
M.FECHA_CANCEL = '01/01/1900' AND 
M.TIPOJUICIO = '' AND 
(CURRENT DATE - E.FECHATRAMITEJUIC) >= 10 AND 
E.FECHA_BAJA = '01/01/1900') T WHERE RN = 1
